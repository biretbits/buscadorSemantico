create table usuario(
cod_usuario int not null auto_increment primary key,
usuario char(45),
nombre char(45),
ap char(45),
am char(45),
telefono int,
gmail  char(50),
contrasena char(200)
);

create table area(
cod_area int not null primary key auto_increment,
nombre_area char(45),
direccion_area char(45),
telefono_area char(45));


create table carrera(
cod_carrera int not null auto_increment ,
nombre_carrera char(100),
direccion_carrera char(200),
cod_area int,
primary key(cod_carrera),
estado char(15),
foreign key(cod_area)references area(cod_area)on delete restrict on update restrict
);



create table estudiante_perdio(
cod_p int not null auto_increment,
estado_ano char(45),
abandono char(45),
cod_grado int,
cod_docente int,
cod_carrera int,
cod_es int,
cod_area int,
fecha date,
primary key(cod_p),
foreign key(cod_grado)references grado(cod_grado)on delete restrict on update restrict,
foreign key(cod_docente)references docente(cod_docente)on delete restrict on update restrict,
foreign key(cod_carrera,cod_es)references estudiante(cod_carrera,cod_es)on delete restrict on update restrict,
foreign key(cod_area)references area(cod_area)on delete restrict on update restrict);


create table plan_de_estudio(
cod_pe int not null auto_increment,
nombre_pe char(200),
ano_pe year,
fecha_pe date,
cod_carrera int,
cod_area int,
estado char(15),
primary key(cod_pe,cod_carrera),
foreign key(cod_carrera)references carrera(cod_carrera)on delete restrict on update restrict,
foreign key(cod_area)references area(cod_area)on delete restrict on update restrict
);


create table grado(
cod_grado int not null auto_increment primary key,
nombre_grado char(50),
estado char(15)
);


create table modalidad_titulacion(
cod_mt int not null auto_increment,
tipo_mt char(200),
cod_pe int,
cod_carrera int,
cod_area int,
fecha date,
estado char(15),
primary key(cod_mt, cod_pe, cod_carrera),
foreign key(cod_pe, cod_carrera)references plan_de_estudio(cod_pe, cod_carrera)on delete restrict on update restrict,
foreign key(cod_area)references area(cod_area)on delete restrict on update restrict
);

create table tipo_asignatura(
cod_tp_asig int not null auto_increment primary key,
asignatura_es char(12)
);

create table asignatura(
cod_asig int not null auto_increment,
sigla_asig char(50),
nombre_asig char(100),
ht int,
hl int,
th int,
pre_req char(50),
cod_pe int,
cod_carrera int,
cod_grado int,
cod_area int,
cod_tp_asig int,
estado char(15) DEFAULT 'activo',
embedding LONGBLOB DEFAULT '',
primary key(cod_asig,cod_pe,cod_carrera,cod_grado),
foreign key(cod_pe,cod_carrera)references  plan_de_estudio(cod_pe,cod_carrera)on delete restrict on update restrict,
 foreign key(cod_grado)references grado(cod_grado)on delete restrict on update restrict,
 foreign key(cod_area)references area(cod_area)on delete restrict on update restrict,
 foreign key(cod_tp_asig)references tipo_asignatura(cod_tp_asig)on delete restrict on update restrict
);


create table docente(
cod_docente int not null auto_increment,
nombre_docente char(50),
ap_docente char(50),
am_docente char(50),
ci_docente int,
profesion char(50),
telefono int,
pais char(50),
departamento char(100),
ciudad char(50),
sexo char(50),
cod_area int,
cod_carrera int,
fecha date,
estado char(15),
primary key(cod_docente),
foreign key(cod_carrera)references carrera(cod_carrera)on delete restrict on update restrict,
foreign key(cod_area)references area(cod_area)on delete restrict on update restrict
);



create table estudiante(
cod_es int not null auto_increment,
nombre_es char(50),
ap_es char(50),
am_es char(50),
titulo_bachiller char(10),
ci int,
pais_es char(50),
departamento char(50),
provincia char(100),
ciudad char(50),
region char(50),
sexo char(50),
cod_area int,
cod_carrera int,
cod_grado int,
estado char(20),
fecha date,
primary key(cod_es),
foreign key(cod_carrera)references carrera(cod_carrera)on delete restrict on update restrict,
foreign key(cod_area)references area(cod_area)on delete restrict on update restrict,
foreign key(cod_grado)references grado(cod_grado)on delete restrict on update restrict
);

create table transferencia(
cod_transferencia int not null auto_increment,
transferencia char(10),
universidad_transferencia char(200),
fecha_hora datetime,
fecha date,
cod_es int,
cod_area int,
cod_carrera int,
estado char(20),
primary key(cod_transferencia,cod_es),
foreign key(cod_es)references estudiante(cod_es)on delete restrict on update restrict
,
foreign key(cod_area)references area(cod_area)on delete restrict on update restrict,
foreign key(cod_carrera)references carrera(cod_carrera)on delete restrict on update restrict
);

create table dicta_asignatura(
cod_dicta int not null auto_increment,
porcentaje_avance int,
ano_dicta year,
cod_docente int,
cod_asig int,
cod_pe int,
cod_carrera int,
cod_grado int,
cod_area int,
estado char(15),
fecha date,
primary key(cod_dicta, cod_docente, cod_asig, cod_pe, cod_carrera,cod_grado),

foreign key(cod_asig,cod_pe,cod_carrera,cod_grado) references asignatura
(cod_asig,cod_pe,cod_carrera,cod_grado)on delete restrict on update restrict,

create table cursa_asignatura(
cod_cursa int not null auto_increment,
ano year,
calificacion int,
estado_asignatura char(50),
desercion char(20),
cod_es int,
cod_dicta int,
cod_docente int,
cod_asig int,
cod_pe int,
cod_carrera int,
cod_grado int,
cod_area int,
estado char(15),
fecha date,
primary key(cod_cursa,cod_es,cod_dicta,cod_docente,cod_asig,cod_pe,cod_carrera,cod_grado),
foreign key(cod_es)references estudiante(cod_es)on delete restrict on update restrict,
foreign key(cod_dicta,cod_docente,cod_asig,cod_pe,cod_carrera,cod_grado)references dicta_asignatura(cod_dicta,cod_docente,cod_asig,cod_pe,cod_carrera,cod_grado)on delete restrict on update restrict
,
foreign key(cod_area)references area(cod_area)on delete restrict on update restrict
);


create table defensa(
cod_def int not null auto_increment,
estado_def char(20),
fecha date,
hora time,
cod_es int,
cod_mt int,
cod_pe int,
cod_carrera int,
cod_area int,
estado char(15),
primary key(cod_def,cod_es,cod_mt,cod_pe,cod_carrera),
foreign key(cod_es)references estudiante(cod_es)on delete restrict on update restrict,
foreign key(cod_mt,cod_pe,cod_carrera)references modalidad_titulacion(cod_mt,cod_pe,cod_carrera)on delete restrict on update restrict
,
foreign key(cod_area)references area(cod_area)on delete restrict on update restrict
);



create table titulado(
cod_titu int not null auto_increment,
estado_titu char(20),
nota_titu int,
cod_def int,
cod_es int,
cod_mt int,
cod_pe int,
cod_carrera int,
cod_area int,
fecha date,
hora time,
estado char(15),
primary key(cod_titu,cod_def,cod_es,cod_mt,cod_pe ,cod_carrera),
foreign key(cod_def,cod_es,cod_mt,cod_pe ,cod_carrera)references defensa(cod_def,cod_es,cod_mt,cod_pe ,cod_carrera)on delete restrict on update restrict
,
foreign key(cod_area)references area(cod_area)on delete restrict on update restrict
);


create table transferir(
cod_transferir int not null auto_increment,
transferir char(10),
universidad_trans char(200),
fecha_hora datetime,
fecha date,
hora time,
cod_cursa int,
cod_es int,
cod_dicta int,
cod_docente int,
cod_asig int,
cod_pe int,
cod_carrera int,
cod_grado int,
cod_area int,
estado char(15),
primary key(cod_transferir,cod_cursa,cod_es,cod_dicta,cod_docente,cod_asig,cod_pe,cod_carrera,cod_grado),
foreign key(cod_cursa,cod_es,cod_dicta,cod_docente,cod_asig,cod_pe,cod_carrera ,
cod_grado)references cursa_asignatura(cod_cursa,cod_es,cod_dicta,cod_docente,cod_asig,cod_pe,cod_carrera ,
cod_grado)on delete restrict on update restrict
);



create table respuesta(
  cod_respuesta int not null auto_increment primary key,
  resp TEXT,
  consulta TEXT,
  descripcion TEXT
);
CREATE TABLE embeddings (
    id INT AUTO_INCREMENT PRIMARY KEY,
    texto TEXT,
    embedding LONGBLOB,
    cod_respuesta int,
    estado char(10) DEFAULT '',
    foreign key(cod_respuesta)references respuesta(cod_respuesta)on delete restrict on update restrict
